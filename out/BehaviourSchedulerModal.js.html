import {
  Dialog,
  DialogActions,
  DialogContent,
  DialogTitle,
  Grid,
  Typography,
  TextField as MuiTextField,
} from "@material-ui/core";
import React, { useState, useEffect } from "react";
import Button from "../common/Button";
import SelectField from "../common/SelectField";
import TextField from "../common/TextField";
import axios from "axios";

 /** * Represents Add/Edit Behaviour Scheduler of the Robot
 * @module {function} Add/Edit Behaviour Scheduler  */


const BehaviourSchedulerModal = ({
  data,
  robot_id,
  setData,
  open,
  handleClose,
  title,
  options,
  handleSave,
}) => { 
  
  const handleChange = (e) => {
    data[e.target.name] = e.target.value;
    console.log(e.target.value)
    setData({ ...data });
  };

  const handleDelete = () => {
    setData({
      name: "",
      behaviour_id: "",
      repeat_type: "",
      date: null,
      time: null,
    });
    handleClose();
  };

  return (
    <>
      <Dialog open={open}>
        <DialogTitle>{title}</DialogTitle>
        <Grid container style={{ width: "600px" }}>
          <Grid item xs={12}>
            <Grid container style={{ marginTop: "10px" }}>
              <Grid
                item
                xs={3}
                style={{ paddingLeft: "10px", paddingTop: "10px" }}
              >
                <Typography variant="subtitle1">Name</Typography>
              </Grid>
              <Grid item xs={9}>
                <TextField
                  type="text"
                  style={{ width: "400px" }}
                  name="name"
                  value={data.name}
                  onChange={handleChange}
                />
              </Grid>
            </Grid>
          </Grid>

          <Grid item xs={12}>
            <Grid container style={{ marginTop: "10px" }}>
              <Grid
                item
                xs={3}
                style={{ paddingLeft: "10px", paddingTop: "10px" }}
              >
                <Typography variant="subtitle1">Behaviour</Typography>
              </Grid>
              <Grid item xs={9}>
                <SelectField
                  style={{ width: "400px" }}
                  name="behaviour_id"
                  options={options}
                  selectValue={data.behaviour_id}
                  onChange={handleChange}
                />
              </Grid>
            </Grid>
          </Grid>

          <Grid
            item
            xs={12}
            style={{ paddingLeft: "10px", paddingTop: "10px" }}
          >
            <Grid container style={{ marginTop: "10px" }}>
              <Grid item xs={3}>
                <Typography variant="subtitle1">Schedule</Typography>
              </Grid>
              <Grid item xs={5}>
                <TextField
                  id="date"
                  name="date"
                  label="Date"
                  type="date"
                  value={data.date}
                  onChange={handleChange}
                  style={{
                    // marginLeft: "8px",
                    // marginRight: "8px",
                    width: 200,
                  }}
                  InputLabelProps={{
                    shrink: true,
                  }}
                />
                {/* </form> */}
              </Grid>
              <Grid item xs={4}>
                <TextField
                  name="time"
                  label="Time"
                  type="time"
                  style={{
                    width: 150,
                    marginLeft: "10px",
                    paddingRight: "30px",
                  }}
                  value={data.time}
                  onChange={handleChange}
                  InputLabelProps={{
                    shrink: true,
                  }}
                  inputProps={{
                    step: 300, // 5 min
                  }}
                />
              </Grid>
              {/* <Grid item xs={3}>
                <SelectField
                  style={{ width: "100px" }}
                  name="retryperiod"
                  label="retry period"
                  options={{ 1: "1", 2: "2" }}
                  selectValue={data.retryperiod}
                  onChange={handleChange}
                />
              </Grid> */}
            </Grid>
          </Grid>
          <Grid
            item
            xs={12}
            style={{ paddingLeft: "10px", paddingTop: "10px" }}
          >
            <Grid container style={{ marginTop: "10px" }}>
              <Grid item xs={3}>
                <Typography variant="subtitle1">Repeat</Typography>
              </Grid>
              <Grid item xs={9}>
                <SelectField
                  style={{ width: "200px" }}
                  name="repeat_type"
                  label="Repeat"
                  options={{
                    "None": "None",
                    "Every Day": "Every Day",
                    "Every Week": "Every Week",
                    "Every Month": "Every Month",
                  }}
                  selectValue={data.repeat_type}
                  onChange={handleChange}
                />
              </Grid>
            </Grid>
          </Grid>
        </Grid>
        <DialogActions style={{ marginTop: "30px" }}>
          <div style={{ flex: "1" }} />
          <Button name={"Discard"} handleClick={handleDelete} />
          <Button
            name={"Save "}
            handleClick={handleSave}
            disabled={!data.name.length}
          />
          <div style={{ flex: "1" }} />
        </DialogActions>
      </Dialog>
    </>
  );
};

export default BehaviourSchedulerModal;
